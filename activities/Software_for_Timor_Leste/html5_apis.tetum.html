  1. Implementa Geolocation API:
    Aumenta kódigu JavaScript atu hetan lokalizasaun uzuáriu nian ho Geolocation API.
    Exemplu: 
    function getLocation() {
    if (navigator.geolocation) {
        navigator.geolocation.getCurrentPosition(showPosition);
    } else {
        document.getElementById("location").innerHTML = "Geolocation la suporta husi browser ne'e.";
    }
}

function showPosition(position) {
    document.getElementById("location").innerHTML = "Latitude: " + position.coords.latitude +
    "<br>Longitude: " + position.coords.longitude;
}
  2. Implementa Armazenamentu Lokal API:
    Aumenta kódigu JavaScript atu salva no hetan dadus ho Armazenamentu Lokal API.
    Exemplu: 
    function saveName() {
    var name = document.getElementById("name").value;
    localStorage.setItem("name", name);
    document.getElementById("savedName").innerHTML = "Naran nebe salva: " + name;
}

// Hetan naran salva iha momentu karga pájina
window.onload = function() {
    var name = localStorage.getItem("name");
    if (name) {
        document.getElementById("savedName").innerHTML = "Naran nebe salva: " + name;
    }
};
  3. Implementa Rasta no Tuir API:
    Aumenta kódigu JavaScript atu habilita funsionalidade rasta no tuir.
    Exemplu: 
    function allowDrop(ev) {
    ev.preventDefault();
}

function drag(ev) {
    ev.dataTransfer.setData("text", ev.target.id);
}

function drop(ev) {
    ev.preventDefault();
    var data = ev.dataTransfer.getData("text");
    ev.target.appendChild(document.getElementById(data));
}
